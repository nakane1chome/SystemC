298 inline
299 sc_logic_value_t
300 sc_bv_base::get_bit( int i ) const
301 {
302  int wi = i / SC_DIGIT_SIZE;
303  int bi = i % SC_DIGIT_SIZE;
304  return sc_logic_value_t( (m_data[wi] >> bi) & SC_DIGIT_ONE );
305 }
306 
307 inline
308 void
309 sc_bv_base::set_bit( int i, sc_logic_value_t value )
310 {
311  int wi = i / SC_DIGIT_SIZE;
312  int bi = i % SC_DIGIT_SIZE;
313  sc_digit mask = SC_DIGIT_ONE << bi;
314  m_data[wi] |= mask; // set bit to 1
315  m_data[wi] &= value << bi | ~mask;
316 }
