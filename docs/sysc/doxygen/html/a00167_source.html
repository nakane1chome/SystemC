112 scfx_string::resize( std::size_t i )
113 {
114  do {
115  m_alloc *= 2;
116  } while( i >= m_alloc );
117 
118  char* temp = new char[m_alloc];
119 
120  for( int j = 0; j < (int) m_len; ++ j ) {
121  temp[j] = m_buffer[j];
122  }
123  temp[m_len] = 0;
124 
125  delete [] m_buffer;
126  m_buffer = temp;
127 }
