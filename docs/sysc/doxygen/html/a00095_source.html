113  static sc_logic_value_t to_value( sc_logic_value_t v )
114  {
115  if( v < Log_0 || v > Log_X ) {
116  invalid_value( v );
117  }
118  return v;
119  }
120 
121  static sc_logic_value_t to_value( bool b )
122  { return ( b ? Log_1 : Log_0 ); }
123 
124  static sc_logic_value_t to_value( char c )
125  {
126  sc_logic_value_t v;
127  unsigned int index = (int)c;
128  if ( index > 127 )
129  {
130  invalid_value(c);
131  v = Log_X;
132  }
133  else
134  {
135  v = char_to_logic[index];
136  if( v < Log_0 || v > Log_X ) {
137  invalid_value( c );
138  }
139  }
140  return v;
141  }

144  {
