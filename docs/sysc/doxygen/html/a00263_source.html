50 template <class T>
51 inline
52 const T
53 sc_abs( const T& a )
54 {
55  // return ( a >= 0 ? a : -a );
56  // the code below is functionaly the same as the code above; the
57  // difference is that the code below works for all arithmetic
58  // SystemC datatypes.
59  T z( a );
60  z = 0;
61  if( a >= z ) {
62  return a;
63  } else {
64  T c( a );
65  c = -a;
66  return c;
67  }
