122 template <class IF>
123 inline
124 const sc_event&
125 sc_event_finder_t<IF>::find_event( sc_interface* if_p ) const
126 {
127  const IF* iface = ( if_p ) ? DCAST<const IF*>( if_p ) :
128  DCAST<const IF*>( port().get_interface() );
129  if( iface == 0 ) {
130  report_error( SC_ID_FIND_EVENT_, "port is not bound" );
131  }
132  return (CCAST<IF*>( iface )->*m_event_method) ();
133 }
