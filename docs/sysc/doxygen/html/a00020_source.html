117 template<class IF>
118 class sc_export : public sc_export_base
119 {

122 public: // constructors:

126  }
127 
128  explicit sc_export( const char* name_ ) : sc_export_base(name_)
129  {
130  m_interface_p = 0;
131  }
132 
133 public: // destructor:
134  virtual ~sc_export()
135  {
136  }
137 
138 public: // interface access:
139 
140  virtual sc_interface* get_interface()
141  {
142  return m_interface_p;
143  }
144 
145  virtual const sc_interface* get_interface() const
146  {
147  return m_interface_p;
148  }
149 
150  const IF* operator -> () const {
151  if ( m_interface_p == 0 )
152  {
153  SC_REPORT_ERROR(SC_ID_SC_EXPORT_HAS_NO_INTERFACE_,name());
154  }
155  return m_interface_p;
156  }
157 
158  IF* operator -> () {
159  if ( m_interface_p == 0 )
160  {
161  SC_REPORT_ERROR(SC_ID_SC_EXPORT_HAS_NO_INTERFACE_,name());
162  }
163  return m_interface_p;
164  }
165 
166  operator IF& ()
167  {
168  if ( m_interface_p == 0 )
169  {
170  SC_REPORT_ERROR(SC_ID_SC_EXPORT_HAS_NO_INTERFACE_,name());
171  }
172  return *m_interface_p;
173  }
174  operator const IF&() const
175  { return *const_cast<this_type*>(this); }
176 
177 
178 public: // binding:
179  SC_VIRTUAL_ void bind( IF& interface_ )
180  {
181  if ( m_interface_p )
182  {
183  SC_REPORT_ERROR(SC_ID_SC_EXPORT_ALREADY_BOUND_,name());
184  }
185  else
186  {
187  m_interface_p = &interface_;
188  }
189  }
190 
191  void operator () ( IF& interface_ )
192  {
193  this->bind(interface_);
194  }
195 
196 public: // identification:
197  virtual const char* kind() const { return "sc_export"; }
198 
199 protected:

201  return typeid( IF ).name();
202  }
