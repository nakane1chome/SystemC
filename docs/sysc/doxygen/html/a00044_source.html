539 template <class IF>
540 inline
541 void
542 sc_port_b<IF>::add_interface( sc_interface* interface_ )
543 {
544  IF* iface = DCAST<IF*>( interface_ );
545  assert( iface != 0 );
546 
547  // make sure that the interface is not already bound:
548 
549  int size = m_interface_vec.size();
550  for ( int i = 0; i < size; i++ )
551  {
552  if ( iface == m_interface_vec[i] )
553  {
554  report_error( SC_ID_BIND_IF_TO_PORT_,
555  "interface already bound to port" );
556  }
557  }
558 
559  // "bind" the interface and make sure our short cut for 0 is set up.
560 
561  m_interface_vec.push_back( iface );
562  m_interface = m_interface_vec[0];
563 }
