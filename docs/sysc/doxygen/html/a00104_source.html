1151  }

1159  set_words_( x, i, x.get_word( i - wn ), x.get_cword( i - wn ) );
1160  }

1162  set_words_( x, i, SC_DIGIT_ZERO, SC_DIGIT_ZERO );
1163  }
1164  }

1166  // shift bits
