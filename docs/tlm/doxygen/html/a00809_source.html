181  }
182  // No debug support
183  return 0;

185

189  if (m_get_direct_mem_ptr) {
190  // forward call
191  sc_assert(m_mod);
192  return (m_mod->*m_get_direct_mem_ptr)(trans, dmi_data);
193  }
194  // No DMI support

197  dmi_data.set_end_address((sc_dt::uint64)-1);
198  return false;
199  }

206  GetDirectMem_ptr m_get_direct_mem_ptr;
207  };
208
